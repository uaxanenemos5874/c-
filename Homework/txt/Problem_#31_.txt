#include <iostream>
#include <cmath>

using namespace std;

int main () {
    int number;

    cout << "Enter number: ";
    cin >> number;

    cout << "Number^2 = " << round(pow(number, 2)) << endl << "Number^3 = " << round(pow(number, 3)) << endl << "Number^4 = " << round(pow(number, 4)) << endl;

    return 0;

}
